\section*{Problem 1}
设$R_1$和$R_2$是集合$A$上的关系并由以下矩阵表示。\\
\centerline{$M_{R_1} = \begin{bmatrix} 0 & 1 &0 \\ 1 & 1 & 1\\ 1 & 0 & 0  \end{bmatrix}$，$M_{R_2} = \begin{bmatrix} 0 & 1 &0 \\ 0 & 1 & 1\\ 1 & 1 & 1  \end{bmatrix}$}
求表示下述关系的矩阵。
\begin{enumerate}[a)]
\item $R_1\cup R_2$
\item $R_1\cap R_2$
\item $R_2\circ R_1$
\item $R_1\circ R_1$
\item $R_1\oplus R_2$
\end{enumerate}

\section*{Problem 4}

\section*{Problem 5}
使用沃舍尔算法找出下列关系的传递闭包。
\begin{enumerate}[a)]
\item $\{(b,c),(b,e),(c,e),(d,a),(e,b),(e,c)\}$
\item $\{(a,e),(b,a),(b,d),(c,d),(d,a),(d,c),(e,a),(e,b),(e,c),(e,e)\}$
\end{enumerate}

\section*{Problem 6}
设$R$是正整数的有序对集合上的关系，$((a,b),(c,d))\in R$ 当且仅当$a+d=b+c$。证明$R$是等价关系。

\section*{Problem 7}
设$R$是所有二进位串的集合上的关系，$sRt$当且仅当$s$和$t$包含相同个数的1，证明$R$是等价关系。

\section*{Problem 8}
当$n$为下列各数时，同余类$[n]_5$（即$n$关于模$5$同余的等价类）是什么？
\begin{enumerate}[a)]
\item $2$
\item $3$
\item $6$
\item $-3$
\end{enumerate}

\section*{Problem 9}
下面哪些子集族是$\{1,2,3,4,5,6\}$的划分？
\begin{enumerate}[a)]
\item $\{1,2\},\{2,3,4\},\{4,5,6\}$
\item $\{1\},\{2,3,6\},\{4\},\{5\}$
\item $\{2,4,6\},\{1,3,5\}$
\item $\{1,4,5\},\{2,6\}$
\end{enumerate}

\section*{Problem 10}
假设$R_1$和$R_2$是集合$A$上的等价关系，$P_1$和$P_2$分别是对应于$R_1$和$R_2$的划分。证明 $R_1 \subseteq R_2$，当且仅当$P_1$是$P_2$的加细。

\section*{Problem 11}
令$R$是有限非空集合$A$上的一个等价关系，试证明：\\
\centerline{ $|R|\cdot|A/R| \ge |A|^2$ } \\
$($ 此处$A/R$是集合$A$依据$R$做划分而得到的商集。$)$
